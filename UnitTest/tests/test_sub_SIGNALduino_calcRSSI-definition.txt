defmod test_sub_SIGNALduino_calcRSSI UnitTest dummyDuino ( 
{
	use Test2::V0;
	use Test2::Tools::Compare qw{is};

	subtest 'Test SIGNALduino_calcRSSI' => sub {
		plan(3);
		my $rssiInput=0;
		my ($rssiStr)=SIGNALduino_calcRSSI($rssiInput);
		is($rssiStr,"RSSI = -74","check return RSSI string for input $rssiInput");

		$rssiInput=83;
		$rssiStr=SIGNALduino_calcRSSI($rssiInput);
		is($rssiStr,"RSSI = -32.5","check return RSSI string for input $rssiInput");

		$rssiInput=228;
		$rssiStr=SIGNALduino_calcRSSI($rssiInput);
		is($rssiStr,"RSSI = -88","check return RSSI string for input $rssiInput");
	};
}
)
