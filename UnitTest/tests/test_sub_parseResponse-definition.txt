defmod test_sub_parseResponse UnitTest dummyDuino ( 
{
 	note("read test from test_sub_parseResponse-definition.txt");

	subtest 'Test cmd "cmds"' => sub {
		plan tests => 1;
		
		my $ret=SIGNALduino_parseResponse($targetHash,"cmds","$target cmds => .*Use one of V R t X S P C r W s x e\n");
		is($ret," V R t X S P C r W s x e","check return value");
	}; 

	subtest 'Test cmd "uptime"' => sub {
		plan tests => 1;
		
		my $ret=SIGNALduino_parseResponse($targetHash,"uptime","86410");
		is($ret,"1 00:00:10","check return value");
	}; 

	subtest 'Test cmd "ccregAll"' => sub {
		plan tests => 1;
		
		my $ret=SIGNALduino_parseResponse($targetHash,"ccregAll","ccreg 00: 00  ccreg 10: 57");
		is($ret,"\n\nccreg 00: 00\nccreg 10: 57","check return value");
	}; 

	subtest 'Test cmd "ccconf"' => sub {
		plan tests => 1;
		
		my $ret=SIGNALduino_parseResponse($targetHash,"ccconf","C0Dn11=10B07157C43023B900070018146C070091");
		is($ret,"freq:433.920MHz bWidth:325KHz rAmpl:42dB sens:8dB  (DataRate:5603.79Baud)","check return value");
	}; 

	subtest 'Test cmd "bWidth"' => sub {
		plan tests => 1;
		$targetHash->{getcmd}->{arg}="325";
		my $ret=SIGNALduino_parseResponse($targetHash,"bWidth","C10 = 57");
		is($ret,"Setting MDMCFG4 (10) to 57 = 325 KHz","check return value");
	}; 
	
	subtest 'Test cmd "ccpatable"' => sub {
		plan tests => 1;
		my $ret=SIGNALduino_parseResponse($targetHash,"ccpatable","C3E = 00 84 00 00 00 00 00 00");
		is($ret,"C3E = 00 84 00 00 00 00 00 00 => 5_dBm","check return value");
	}; 

	subtest 'Test not specified cmd (default behaviour)' => sub {
		plan tests => 1;
		my $ret=SIGNALduino_parseResponse($targetHash,"something","some message to verify\n");
		is($ret,"some message to verify","check return value");
	}; 

} 
)
